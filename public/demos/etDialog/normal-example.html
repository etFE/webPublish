点击下面按钮，通过父页面，全屏打开一个页面

// 通过父页面 全屏打开一个子页面
parent.$.etDialog.open({
  url: './max-example.html',
  isMax: true,
  frameName: window.name, // 重要参数，必写
  title: '全屏的子页面',
})
// 在当前子页面中关闭自身窗口
var curIndex = parent.$.etDialog.getFrameIndex(window.name);
parent.$.etDialog.close(curIndex);

执行父页面的方法，或者拿取父页面的数据，父页面需要全局暴露

// 执行父页面的查询方法
parent.query();
// 获取父页面全局的grid参数，并进行相关操作
parent.grid.getAllData();
